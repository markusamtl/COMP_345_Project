453
454 for (Player* p : players) {
455
456 if(p == nullptr || p -> getPlayerName() == "Neutral"){ continue; } //Skip null players and neutral player
457 p -> issueOrder(); //Randomly issue an order
458
459 //Add random logic to create a real order. Will help to simulate gameplay for now.
460 Order* issuedOrder = p -> getPlayerOrders() -> peek();
461
462 //Get random numbers to help populate order info
463 int randomNumberOne = static_cast<int>(TimeUtil::getSystemTimeNano());
464 int randomNumberTwo = static_cast<int>(TimeUtil::getSystemTimeNano());
465 int randomNumberThree = static_cast<int>(TimeUtil::getSystemTimeNano());
466
467 switch(issuedOrder -> getOrderType()) {
468
469 case(OrderType::Advance):{ //Player advances to a random territory
470
471 vector<Territory*> ownedTerrs = p -> getOwnedTerritories().getTerritories(); //Get all of the player's territories
472 if(ownedTerrs.empty()){ break; } //Should never happen, but just in case

474 Territory* sourceTerr = ownedTerrs[randomNumberOne % ownedTerrs.size()]; //Random owned territory

478 Territory* targetTerr = neighbors[randomNumberTwo % neighbors.size()]; //Random neighbor territory
479 Player* targetPlayer = targetTerr -> getOwner(); //Get owner of that territory
480 if(targetPlayer == nullptr){ break; } //Should never happen, but just in case

482 int numArmiesToMove = (randomNumberThree % sourceTerr -> getNumArmies()); //At least 1 army, at most all armies

484
485 Advance* advanceOrder = new Advance(p, sourceTerr, targetTerr, numArmiesToMove);

488
489 OrderList* playerOrders = p -> getPlayerOrders();
490 if(playerOrders == nullptr){ break; } //Should never happen, but just in case

493
494 }
495
496 } break;

499

503 Territory* randomTarget = candidateTargets[randomNumberOne % candidateTargets.size()]; //Random target territory

509

617 string GameEngine::executeOrder() {
618
619 if(state != EngineState::ExecuteOrders){ return "Invalid state transition: must be in ExecuteOrders."; }
620
621 // ---------------------------- Setup ----------------------------
622 vector<Player*> ordered;
623 queue<Player*> tempQueue = playerQueue;
624
625 while (!tempQueue.empty()){ // Transfer to vector for easier iteration
626
627 ordered.push_back(tempQueue.front());
628 tempQueue.pop();
629
630 }
631
632 // Vector to collect players that must be deleted later
633 vector<Player*> toDelete;
634
635 // ---------------------------- Execute Orders ----------------------------
636 for(Player* p : ordered) {
637
638 if(p == nullptr || p -> getPlayerName() == "Neutral"){ continue; } //Skip null players and neutral player
639
640 cout << "[" << p -> getPlayerName() << "] executing orders..." << endl;
641
642 OrderList* orders = p -> getPlayerOrders();
643
644 if(!orders || orders->size() == 0) { //No orders to execute
645 cout << "No orders to execute for " << p -> getPlayerName() << endl;
646 continue;
647 }
648
649 while(orders -> size() > 0) { //Execute all orders for this player
650
651 Order* o = orders-> peek(); //Get the first order
652
653 if(o != nullptr) {
654
655 cout << "[" << p -> getPlayerName() << "] Executes: " << *o << endl;
656 o -> execute();
657
658 }
659
660 orders -> removeOrder(0);
661 }
662
663 cout << "[" << p -> getPlayerName() << "] has finished executing orders." << endl;
664
665 }
666
667 // ---------------------------- Eliminate Players ----------------------------
668 for(auto it = players.begin(); it != players.end();) { //Use an iterator to allow safe removal
669
670 Player* p = *it; //Dereference iterator to get player
671
672 if(p == nullptr || p->getPlayerName() == "Neutral") {
673 it++;
674 continue;
675 }
676
677 int terrCount = static_cast<int>(p->getOwnedTerritories().size());
678 if(terrCount == 0) {
679
680 cout << "[" << p -> getPlayerName() << "] has been eliminated from the game." << endl;
681
682 // Remove from playerQueue
683 queue<Player*> newQueue;
684
685 while(!playerQueue.empty()) {
686
687 Player* front = playerQueue.front(); //Peek at front of queue
688 playerQueue.pop(); //Remove from old queue
689 if(front != p){ newQueue.push(front); } //Only add to new queue if not the eliminated player
690
691 }
692
693 playerQueue = newQueue;
694
695 // Mark player for deletion AFTER this method finishes
696 toDelete.push_back(p);
697 it = players.erase(it); //Remove from players list and get new iterator
698 continue;
699 }
700
701 it++;
702
703 }
704
705 // ---------------------------- Win Condition Check ----------------------------
706 const unordered_map<Continent*, long long>& refTable = gameMap -> getContinentLookupTable(); //Get reference table from map
707
708 for(Player* p : ordered) {
709
710 // Skip eliminated or null players
711 if(p == nullptr){ continue; }
712
713 if(find(toDelete.begin(), toDelete.end(), p) != toDelete.end()){ continue; }//Skip players marked for deletion
714
715 if(p -> hasWon(refTable)) { //Only ONE Winner
716
717 currentPlayer = p;
718 state = EngineState::Win;
719
720 // Log the winner
721 cout << "\n=== WINNER DETECTED ===" << endl;
722 cout << "[" << p -> getPlayerName() << "] controls all continents!" << endl;
723 return "win";
724
725 }
726
727 }
728
729 // ---------------------------- Deferred Cleanup ----------------------------
730 for(Player* dead : toDelete) {
731
732 if(dead != nullptr) {
733
734 cout << "[Cleanup] Deleting player object for " << dead->getPlayerName() << endl;
735 delete dead;
736
737 }
738
739 }
740
741 toDelete.clear();
742
743 // ---------------------------- Phase Transition ----------------------------
744 state = EngineState::AssignReinforcement;
745 cout << "Assignment Reinforcement phase has been completed, moving on to next phase" << endl;
746 return "endexecuteorder";
