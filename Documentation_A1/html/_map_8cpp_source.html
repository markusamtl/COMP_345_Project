303 bool Territory::territoryIDCompare(Territory* a, Territory* b) {
304
305 string idA = a -> getID();
306 string idB = b -> getID();
307
308 //This method thinks ahead for cases like AA10 vs A10 (A10 should come first)
309
310 size_t i = 0, j = 0; //Keep track of indexes in ID string
311
312 while(i < idA.size() && j < idB.size()) {
313
314 if(std::isdigit(idA[i]) && std::isdigit(idB[j])) { //Extract full number substrings
315
316 size_t iStart = i, jStart = j;
317
318 while(i < idA.size() && isdigit(idA[i])){ i++; } //Iterate over the numerical values of the first ID
319 while(j < idB.size() && isdigit(idB[j])){ j++; } //Iterate over the numerical values of the first ID
320
321 int numA = stoi(idA.substr(iStart, i - iStart)); //Cast the number
322 int numB = stoi(idB.substr(jStart, j - jStart)); //Cast the number
323
324 if(numA != numB) { return numA < numB; }
325
326 } else { //Compare as characters
327
328 if(idA[i] != idB[j]){ return idA[i] < idB[j]; }
329 i++;
330 j++;
331
332 }
333
334 }
335
336 //If one string is a prefix of the other, shorter comes first
337 return idA.size() < idB.size();
338

646 bool Map::isMapConnectedDFS(const vector<Territory*>& terrList, Continent* restrictCont) const {
647
648 if(terrList.empty()){ return false; } //If territories are empty, stop
649
650 int totalTerritories = static_cast<int>(terrList.size());
651 unordered_map<Territory*, int> terrIndex; //Set up hashmap, Territory* -> index of territory in terrList
652
653 for (int i = 0; i < totalTerritories; i++) { terrIndex[terrList[i]] = i;} //Map each territory to its index in the hashmap
654
655 vector<bool> visited(totalTerritories, false); //Create a visited array to track visited territories
656 stack<Territory*> toVisit; //Create a stack for DFS traversal. Currently unvisited territories will be pushed onto this stack, popped off once visited
657
658 Territory* start = terrList[0]; //Start from the first territory in the list
659 toVisit.push(start); //Push it onto the stack
660 visited[terrIndex[start]] = true; //Mark it as visited
661
662 while(!toVisit.empty()) { //Iterate through the stack until it's empty
663
664 Territory* curTerr = toVisit.top();
665 toVisit.pop();
666
667 for(Territory* neighbor : curTerr -> getNeighbors()) { //For each neighbor of the current territory
668
669 if(neighbor == nullptr) { continue; } //Skip for empty neighbours
670
671 //If the method is restricted to a continent, skip neighbors outside it
672 if(restrictCont && neighbor->getContinent() != restrictCont) { continue; }
673
674 int neighIndex = terrIndex[neighbor];
675
676 if(!visited[neighIndex]) { //If the neighbor hasn't been visited yet
677
678 visited[neighIndex] = true; //Mark it as visited
679 toVisit.push(neighbor);
680
681 }
682
683 }
684
685 }
686
687 int totalVisited = 0;
688 for (bool v : visited) { if (v) totalVisited++; } // Count visited territories
689
690 return (totalVisited == totalTerritories); //If all territories were visited, the graph is connected
691
692 }

694 bool Map::validate() const {
695
696 // -- 1. Non-empty checks --
697 if(territories.empty()) { //If no territories exist in the map
698
699 cerr << "Validation failed: Map has no territories." << endl;
700 return false;
701
702 }
703
704 if(continents.empty()) { //If no continents exist in the map
705
706 cerr << "Validation failed: Map has no continents." << endl;
707 return false;
708
709 }
710
711 //-- 2. Each territory must belong to a continent --
712 for(Territory* t : territories) { //For each territory in the map
713
714 if(t -> getContinent() == nullptr) { //If the territory does not belong to any continent
715
716 cerr << "Validation failed: Territory '" << t -> getID() << "' does not belong to any continent." << endl;
717 return false;
718
719 }
720
721 }
722
723 //-- 3. Check overall map connectivity --
724 if (!isMapConnectedDFS(territories, nullptr)) { // No continent restriction, perform DFS on the WHOLE map
725
726 cerr << "Validation failed: Map is not fully connected." << endl;
727 return false;
728
729 }
730
731 //-- 4. Check continent connectivity --
732 for(Continent* cont : continents) { // For each continent
733
734 const vector<Territory*>& terrList = cont->getTerritories(); //Get current continent's territories
735
736 if(terrList.empty()) { //If no territories are associated with the continent
737
738 cerr << "Validation failed: Continent '" << cont -> getID() << "' has no territories." << endl;
739 return false;
740
741 }
742
743 if(!isMapConnectedDFS(terrList, cont)) { //Perform DFS for the continent's territories only
744
745 cerr << "Validation failed: Continent '" << cont -> getID() << "' is not a connected subgraph." << endl;
746 return false;
747
748 }
749
750 }
751
752 cout << "Validation successful: Map is valid." << endl;
753 return true;
754
755 }

932 int MapLoader::importMapInfo(const string& filePath) {
933
934 ifstream file(filePath); // Open file buffer on filePath
935
936 if(!file.is_open()) { // If the file directory is invalid
937
938 cerr << "Error: Could not open file " << filePath << endl;
939 return MAP_FILE_NOT_FOUND;
940
941 }
942
943 string line; // To hold each line of the file
944 string section; // To track which section of the map file is being parsed
945
946 while(getline(file, line)) { // Read file line by line
947
948 line = StringHandling::trim(line); // Remove spaces and \r
949 if(line.empty()) continue; // Skip empty lines
950
951 // Detect section headers
952 if(line[0] == '[') {
953 section = line; // Store section name, e.g., "[Map]"
954 continue;
955 }
956
957 // Parse [Map] section
958 if(section == "[Map]") {
959
960 if(line.find("author=") == 0) author = line.substr(7); // Get everything after "author="
961 else if(line.find("image=") == 0) image = line.substr(6); // Get everything after "image="
962 else if(line.find("wrap=") == 0) wrap = line.substr(5); // Get everything after "wrap="
963 else if(line.find("scroll=") == 0) scrollType = line.substr(7); // Get everything after "scroll="
964 else if(line.find("warn=") == 0) warn = line.substr(5); // Get everything after "warn="
965
966 else { // The above are the only valid lines in the [Map] section
967 cerr << "Invalid line in [Map]: " << line << endl;
968 return MAP_PARSE_ERROR;
969 }
970
971 } else if (section == "[Continents]") { // Parse [Continents] section
972
973 size_t pos = line.find('='); // Find position of '=' to split continent name and bonus value
974
975 if(pos == string::npos) { // If there is no '=' character, the line is invalid
976
977 cerr << "Invalid continent line: " << line << endl;
978 return MAP_PARSE_ERROR;
979 }
980
981 string continentName = StringHandling::trim(line.substr(0, pos)); // Extract continent name
982 string continentBonusStr = StringHandling::trim(line.substr(pos + 1)); // Extract continent name
983
984 pair<bool, int> bonusValueStatus = StringHandling::isStrInt(continentBonusStr); //Get bonus valus
985
986 if(bonusValueStatus.first == true) { continents[continentName] = bonusValueStatus.second; }
987 else { return MAP_PARSE_ERROR; }
988
989
990 } else if (section == "[Territories]") { // Parse [Territories] section
991
992 vector<string> fields; //To hold the fields of the territory line
993 string token; //Holds each field temporarily for a territory line
994 stringstream ss(line); //Use stringstream for easy CSV parsing
995
996 while(getline(ss, token, ',')) { //Split line by commas
997
998 string trimmed = StringHandling::trim(token); //Trim leading/trailing spaces
999 if(!trimmed.empty()){ fields.push_back(trimmed); } //Make sure the territory info isn't empty
1000
1001 }
1002
1003 if(fields.size() < 4) { // Must have at least ID, XCoord, YCoord, Continent
1004
1005 cerr << "Invalid territory line: " << line << endl;
1006 return MAP_PARSE_ERROR;
1007
1008 }
1009
1010 territories.push_back(fields); // Add the parsed territory to the territories vector
1011
1012 } else { // If section is invalid
1013
1014 cerr << "Unknown section: " << section << endl;
1015 return MAP_INVALID_SECTION;
1016
1017 }
1018
1019 }
1020
1021 file.close();
1022
1023 cout << "Map loaded successfully!" << endl << " Continents loaded: " << continents.size() << endl << " Territories loaded: " << territories.size() << endl;
1024
1025 return MAP_OK;
1026 }
