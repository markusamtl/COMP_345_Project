301 bool Advance::validate() const {
302
303 //Check if either of the territories, or the player, are null pointers
304 if(issuer == nullptr || source == nullptr || target == nullptr) { return false; }
305
306 //Player must own the source territory
307 if(source -> getOwner() != issuer){ return false; }
308

314
315 //Check if the target territory is a neighbour of the source territory

317 if(!isAdjacent) { return false; }
318
319 //Check if the attacker has a truce with the defender

321
322 if (defender != nullptr) {
323
324 const vector<string>& neutrals = issuer->getNeutralEnemies();
325
326 if(find(neutrals.begin(), neutrals.end(), defender->getPlayerName()) != neutrals.end()) { //Search through list of neutral enemies

332 }

338 void Advance::execute() {
339
340 // --- 1. Validate ownership and adjacency ---
341 if(!this -> validate()) {
342
343 this -> setEffect("Advance order invalid.");
344 cout << this -> getEffect() << endl;
345 return;
346
347 }
348
349 // --- 2. Check if source belongs to issuer ---
350 if(source -> getOwner() != issuer) {
351
352 this -> setEffect("Advance failed: source territory not owned by issuer.");
353 cout << this -> getEffect() << endl;
354 return;
355
356 }
357
358 // --- 3. Check adjacency ---
359 const vector<Territory*>& neighbors = source -> getNeighbors();
360
361 //Check if the target is in the neighbour vector
362 if(find(neighbors.begin(), neighbors.end(), target) == neighbors.end()){
363
364 this -> setEffect("Advance failed: target territory is not adjacent.");
365 cout << this -> getEffect() << endl;
366 return;
367
368 }
369
370 // --- 4. If target is also owned by issuer, reinforce ---
371 if (target->getOwner() == issuer) {
372
373 if (source->getNumArmies() <= 1) { //1 Army MUST remain on the source
374
375 this->setEffect("Advance failed: Not enough armies to advance with.");
376 cout << this->getEffect() << endl;
377 return;
378
379 }
380
381 int numArmiesMovable = min(numArmies, source->getNumArmies() - 1); //Ensure 1 army stays on source
382
383 source->setNumArmies(source->getNumArmies() - numArmiesMovable); // Remove Armies from source
384 target->setNumArmies(target->getNumArmies() + numArmiesMovable); // Add armies to target
385
386 //Ensure both territories are tracked by issuer ---
387 if (!issuer -> getOwnedTerritories().owns(source)) { issuer -> addOwnedTerritories(source); }
388 if (!issuer->getOwnedTerritories().owns(target)) { issuer->addOwnedTerritories(target); }
389
390 this->setEffect("Advance successful: moved " + to_string(numArmiesMovable) +
391 " armies from " + source -> getID() +
392 " to " + target -> getID() + ".");
393
394 cout << this->getEffect() << endl;
395
396 return;
397
398 }
399
400
401 // --- 5. Perform battle logic
402 if(source -> getNumArmies() <= 1) { //1 Army MUST remain on the source
403
404 this -> setEffect("Advance failed: Not enough armies to attack with.");
405 cout << this -> getEffect() << endl;
406 return;
407
408 }
409
410 int attackArmies = min(numArmies, source -> getNumArmies() - 1); //Ensure 1 army stays on source
411 int defenseArmies = target -> getNumArmies();
412
413 //Remove armies from source no matter what
414 source -> setNumArmies(source -> getNumArmies() - attackArmies);
415
416 int attackersRemaining = attackArmies;
417 int defendersRemaining = defenseArmies;
418
419 //Boolean flags to help determine the result of battle-turns
420 bool reduceAttacker = false;
421 bool reduceDefense = false;
422
427 while(attackersRemaining > 1 && defendersRemaining > 0){
428
429 reduceAttacker = false;
430 reduceDefense = false;
431
432 //Defender has 70% chance to kill attacking army
433 if((static_cast<long long>(TimeUtil::getSystemTimeNano()) % 10000) < 7000) {
434
435 reduceAttacker = true;
436
437 }
438
439
440 //Attacker has 60% chance to kill defending army
441 if((static_cast<long long>(TimeUtil::getSystemTimeNano()) % 10000) < 6000){
442
443 reduceDefense = true;
444
445 }
446
447 //Prevents attackers from being reduced to 1 in the case of attacker winning
448 if(reduceAttacker && attackersRemaining > 1){ attackersRemaining--; }
449 if(reduceDefense && defendersRemaining > 0){ defendersRemaining--; }
450
451 }
452
453 //--- 6. Resolve outcome ---
454
455 Player* oldOwner = target->getOwner(); //Get old conquered territory owner
456 string oldOwnerName = (oldOwner ? oldOwner -> getPlayerName() : "Unknown Player");
457
458 if (defendersRemaining <= 0) {
459
460 if (attackersRemaining > 1) { // Successful conquest
461
462 //Transfer ownership
463 issuer->addOwnedTerritories(target);
464 target->setNumArmies(attackersRemaining);
465
466 issuer->setGenerateCardThisTurn(true); // Attacker earns a card at turn end
467
468 this->setEffect("Advance battle successful: " + issuer->getPlayerName() +
469 " defeated " + oldOwnerName +
470 ", and conquered " + target->getID() + " with " +
471 to_string(attackersRemaining) + " surviving armies.");
472
473 } else { // Edge case: attacker “wins” but can’t move in
474
475 defendersRemaining = 1;
476 target->setNumArmies(defendersRemaining);
477
478 this->setEffect("Advance battle inconclusive: " +
479 issuer->getPlayerName() + " tried to beat " + oldOwnerName +
480 ", but didn't have enough armies to move in. The defender clings onto " + target->getID() +
481 " with 1 army, after a desperate defense.");
482
483 }
484
485 } else { // Defender Wins
486
487 target -> setNumArmies(defendersRemaining);
488
489 this->setEffect("Advance battle failed: defender " +
490 oldOwnerName + " beat " + issuer -> getPlayerName() + ", and holds onto"
491 + target -> getID() + " with " +
492 to_string(defendersRemaining) + " armies remaining.");
493
494 }
495
496 cout << this -> getEffect() << endl;
497
498 }
