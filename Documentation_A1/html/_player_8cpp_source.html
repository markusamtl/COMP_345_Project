547 void Player::issueOrder() {
548
549 if(this -> playerOrders == nullptr){ return; }//Make sure the OrderList for the player exists
550
551 //Track which order types are available to be issued by card. Advance is ALWAYS available.
552 vector<pair<bool, int>> orderTypesAvailableByCard = {{true, 0}, {false, 1}, {false, 2}, {false, 3}, {false, 4}};
553
554 if(playerHand != nullptr) {
555
556 // Check for each card type in the player's hand
557 orderTypesAvailableByCard[1].first = playerHand -> hasCardOfType(WarzoneCard::CardType::Bomb);
558 orderTypesAvailableByCard[2].first = playerHand -> hasCardOfType(WarzoneCard::CardType::Blockade);
559 orderTypesAvailableByCard[3].first = playerHand -> hasCardOfType(WarzoneCard::CardType::Airlift);
560 orderTypesAvailableByCard[4].first = playerHand -> hasCardOfType(WarzoneCard::CardType::Diplomacy);
561
562 }
563
564 int randomOrder = -1; //Initialize to invalid order
565
566 // Keep looping until we select a valid order
567 while (randomOrder == -1 && !orderTypesAvailableByCard.empty()) {
568
569 int randomOrderIndex = static_cast<int>(TimeUtil::getSystemTimeNano() % orderTypesAvailableByCard.size());
570 auto orderTypeInfo = orderTypesAvailableByCard[randomOrderIndex];
571
572 if(orderTypeInfo.first) {
573
574 // Valid order type
575 randomOrder = orderTypeInfo.second;
576
577 }
578 else if (orderTypeInfo.second != 0) {
579
580 // Remove invalid entries EXCEPT Advance (index 0 must stay)
581 orderTypesAvailableByCard.erase(orderTypesAvailableByCard.begin() + randomOrderIndex);
582
583 }
584
585 }
586
587 if (randomOrder == -1) {
588 // Safety fallback (should never happen since Advance is always valid)
589 randomOrder = 0;
590 }
591
592 Order* tempOrder = nullptr; //Instantiate a order pointer
593
594 switch(randomOrder) {
595
596 case 0: tempOrder = new WarzoneOrder::Advance(this, nullptr, nullptr, 0); break; //Advance
597 case 1: tempOrder = new WarzoneOrder::Bomb(this, nullptr); break; //Bomb
598 case 2: tempOrder = new WarzoneOrder::Blockade(this, nullptr, nullptr); break; //Blockade
599 case 3: tempOrder = new WarzoneOrder::Airlift(this, nullptr, nullptr, 0); break; //Airlift
600 case 4: tempOrder = new WarzoneOrder::Negotiate(this, nullptr); break; //Negotiate
601
602 }
603
604 if(tempOrder != nullptr) {
605
606 this -> getPlayerOrders() -> addOrder(tempOrder);
607
608 }
609
610 }
